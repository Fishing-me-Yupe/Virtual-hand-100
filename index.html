// ======================= // Texture // ======================= const textureLoader = new THREE.TextureLoader(); const topFingerTexture = textureLoader.load( "./top-finger.svg", () => { console.log( "top-finger.svg loaded" ); }, undefined, (err) => { console.error( "top-finger.svg error", err ); } ); const middleFingerTexture = textureLoader.load( "./middle-finger.svg", () => { console.log( "middle-finger.svg loaded" ); }, undefined, (err) => { console.error( "middle-finger.svg error", err ); } ); // SVGは上下反転しやすいので補正 topFingerTexture.flipY = false; middleFingerTexture.flipY = false; // 色空間補正 topFingerTexture.colorSpace = THREE.SRGBColorSpace; middleFingerTexture.colorSpace = THREE.SRGBColorSpace;
